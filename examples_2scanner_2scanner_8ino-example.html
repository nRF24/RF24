examples/scanner/scanner.ino

Example to detect interference on the various channels available. This is a good diagnostic tool to check whether you're picking a good channel for your application.

Inspired by cpixip. See http://arduino.cc/forum/index.php/topic,54795.0.html

89const int num_reps = 100;
90bool constCarrierMode = 0;

92void loop(void) {
93 /****************************************/
94 // Send g over Serial to begin CCW output
95 // Configure the channel and power level below
96 if (Serial.available()) {
97 char c = Serial.read();
98 if (c == 'g') {
99 constCarrierMode = 1;
100 radio.stopListening();
101 delay(2);
102 Serial.println("Starting Carrier Out");
103 radio.startConstCarrier(RF24_PA_LOW, 40);
104 } else if (c == 'e') {
105 constCarrierMode = 0;
106 radio.stopConstCarrier();
107 Serial.println("Stopping Carrier Out");
108 }
109 }
110 /****************************************/
111
112 if (constCarrierMode == 0) {
113 // Clear measurement values
114 memset(values, 0, sizeof(values));
115
116 // Scan all channels num_reps times
117 int rep_counter = num_reps;
118 while (rep_counter--) {
119 int i = num_channels;
120 while (i--) {
121 // Select this channel
122 radio.setChannel(i);
123
124 // Listen for a little
125 radio.startListening();
126 delayMicroseconds(128);
127 radio.stopListening();
128
129 // Did we get a carrier?
130 if (radio.testCarrier()) {
131 ++values[i];
132 }
133 }
134 }
135
136
137 // Print out channel measurements, clamped to a single hex digit
138 int i = 0;
139 while (i < num_channels) {
140 if (values[i])
141 Serial.print(min(0xf, values[i]), HEX);
142 else
143 Serial.print(F("-"));
144
145 ++i;
146 }
147 Serial.println();
148
149 } //If constCarrierMode == 0
150}
