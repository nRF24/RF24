16 void RF24::csn(bool mode)
17 {
18  #if defined(RF24_TINY)
19  if (ce_pin != csn_pin) {
20  digitalWrite(csn_pin, mode);
21  }
22  else {
23  if (mode == HIGH) {
24  PORTB |= (1<<PINB2); // SCK->CSN HIGH
25  delayMicroseconds(RF24_CSN_SETTLE_HIGH_DELAY); // allow csn to settle.
26  }
27  else {
28  PORTB &= ~(1<<PINB2); // SCK->CSN LOW
29  delayMicroseconds(RF24_CSN_SETTLE_LOW_DELAY); // allow csn to settle
30  }
31  }
32  // Return, CSN toggle complete
33  return;
34 
35  #elif defined(ARDUINO) && !defined(RF24_SPI_TRANSACTIONS)
36  // Minimum ideal SPI bus speed is 2x data rate
37  // If we assume 2Mbs data rate and 16Mhz clock, a
38  // divider of 4 is the minimum we want.
39  // CLK:BUS 8Mhz:2Mhz, 16Mhz:4Mhz, or 20Mhz:5Mhz
40 
41  #if !defined(SOFTSPI)
42  _SPI.setBitOrder(MSBFIRST);
43  _SPI.setDataMode(SPI_MODE0);
44  #if !defined(F_CPU) || F_CPU < 20000000
45  _SPI.setClockDivider(SPI_CLOCK_DIV2);
46  #elif F_CPU < 40000000
47  _SPI.setClockDivider(SPI_CLOCK_DIV4);
48  #elif F_CPU < 80000000
49  _SPI.setClockDivider(SPI_CLOCK_DIV8);
50  #elif F_CPU < 160000000
51  _SPI.setClockDivider(SPI_CLOCK_DIV16);
52  #elif F_CPU < 320000000
53  _SPI.setClockDivider(SPI_CLOCK_DIV32);
54  #elif F_CPU < 640000000
55  _SPI.setClockDivider(SPI_CLOCK_DIV64);
56  #elif F_CPU < 1280000000
57  _SPI.setClockDivider(SPI_CLOCK_DIV128);
58  #else
59  #error "Unsupported CPU frequency. Please set correct SPI divider."
60  #endif
61 
62  #endif
63  #elif defined(RF24_RPi)
64  if(!mode)
65  _SPI.chipSelect(csn_pin);
66  #endif // defined(RF24_RPi)
67 
68  #if !defined(RF24_LINUX)
69  digitalWrite(csn_pin, mode);
70  delayMicroseconds(csDelay);
71  #endif // !defined(RF24_LINUX)
72 }

231 void RF24::write_payload(const void* buf, uint8_t data_len, const uint8_t writeType)
232 {
233  const uint8_t* current = reinterpret_cast<const uint8_t*>(buf);
234 
235  uint8_t blank_len = !data_len ? 1 : 0;
236  if (!dynamic_payloads_enabled) {
237  data_len = rf24_min(data_len, payload_size);
238  blank_len = payload_size - data_len;
239  }
240  else {
241  data_len = rf24_min(data_len, 32);
242  }
243 
244  //printf("[Writing %u bytes %u blanks]",data_len,blank_len);
245  IF_SERIAL_DEBUG(printf("[Writing %u bytes %u blanks]\n", data_len, blank_len); );
246 
247  #if defined(RF24_LINUX)
248  beginTransaction();
249  uint8_t * prx = spi_rxbuff;
250  uint8_t * ptx = spi_txbuff;
251  uint8_t size;
252  size = data_len + blank_len + 1 ; // Add register value to transmit buffer
253 
254  *ptx++ = writeType;
255  while ( data_len-- )
256  *ptx++ = *current++;
257  while ( blank_len-- )
258  *ptx++ = 0;
259 
260  _SPI.transfernb((char *)spi_txbuff, (char *)spi_rxbuff, size);
261  status = *prx; // status is 1st byte of receive buffer
262  endTransaction();
263 
264  #else // !defined(RF24_LINUX)
265 
266  beginTransaction();
267  status = _SPI.transfer(writeType);
268  while (data_len--) {
269  _SPI.transfer(*current++);
270  }
271  while (blank_len--) {
272  _SPI.transfer(0);
273  }
274  endTransaction();
275 
276  #endif // !defined(RF24_LINUX)
277 }

726  // Enabling 16b CRC is by far the most obvious case if the wrong timing is used - or skipped.
727  // Technically we require 4.5ms + 14us as a worst case. We'll just call it 5ms for good measure.
728  // WARNING: Delay is based on P-variant whereby non-P *may* require different timing.
729  delay(5);

734  setRetries(5, 15);

737  // hardware.

741  uint8_t before_toggle = read_register(FEATURE);

748  toggle_features();
749  }
750  // allow use of multicast parameter and dynamic payloads by default
751  write_register(FEATURE, 0);
752  }
753  ack_payloads_enabled = false; // ack payloads disabled by default
754  write_register(DYNPD, 0); // disable dynamic payloads by default (for all pipes)
755  dynamic_payloads_enabled = false;

762  // This channel should be universally safe and not bleed over into adjacent
763  // spectrum.
764  setChannel(76);

767  // Notice reset and flush is the last thing we do
768  write_register(NRF_STATUS, _BV(RX_DR) | _BV(TX_DS) | _BV(MAX_RT));

770

772  flush_rx();
773  flush_tx();

775  // Clear CONFIG register:
776  // Reflect all IRQ events on IRQ pin
777  // Enable PTX
778  // Power Up
779  // 16-bit CRC (CRC required by auto-ack)
780  // Do not write CE high so radio will remain in standby I mode
781  // PTX should use only 22uA of power
782  write_register(NRF_CONFIG, (_BV(EN_CRC) | _BV(CRCO)) );
783  config_reg = read_register(NRF_CONFIG);
784 
785  powerUp();
786 
787  // if config is not set correctly then there was a bad response from module
788  return config_reg == (_BV(EN_CRC) | _BV(CRCO) | _BV(PWR_UP)) ? true : false;

938 bool RF24::writeBlocking(const void* buf, uint8_t len, uint32_t timeout)
939 {

941  //Keep track of the MAX retries and set auto-retry if seeing failures
942  //This way the FIFO will fill up and allow blocking until packets go through
943  //The radio will auto-clear everything in the FIFO as long as CE remains high
944 
945  uint32_t timer = millis(); // Get the time that the payload transmission started
946

956  if (millis() - timer > (timeout + 95)) {
957  errNotify();
958  #if defined(FAILURE_HANDLING)

961  }
962  #endif
963 
964  }

967  startFastWrite(buf, len, 0); // Write the payload if a buffer is clear
968
